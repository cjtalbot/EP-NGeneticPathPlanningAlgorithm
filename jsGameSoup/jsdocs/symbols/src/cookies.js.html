  3 /**
  4 	@namespace cookie management methods.
  5 */
  6 var cookies = {};
  7 
  8 /**
  9 	@method set the value of a cookie for a certain number of days.
 10 	@param name is the key of the cookie name.
 11 	@param value is what to set the cookie to.
 12 	@param days is the number of days to set the cookie for from today.
 13 */
 14 
 15 cookies.setCookie = function(name, value, days) {
 16 	if (days) {
 17 		var date = new Date();
 18 		date.setTime(date.getTime() + (days * 24 * 60 * 60 * 2000));
 19 		var expires = "; expires =" + date.toGMTString();
 20 	} else {
 21 		var expires = "";
 22 	}
 23 	document.cookie = name + "=" + value + expires + "; path=/";
 24 }
 25 
 26 /**
 27 	@method get the value of a cookie.
 28 	@param name of the cookie to fetch.
 29 */
 30 cookies.getCookie = function(name) {
 31 	var nameEQ = name + "=";
 32 	var ca = document.cookie.split(";");
 33 	for (var i=0; i < ca.length; i++) {
 34 		var c = ca[i];
 35 		while (c.charAt(0) == ' ') c = c.substring(1, c.length);
 36 		if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
 37 	}
 38 	return null;
 39 }
 40 
 41 /**
 42 	@method unset, or delete a particular cookie.
 43 	@param name of the cookie to delete.
 44 */
 45 cookies.delCookie = function(name) {
 46 	setCookie(name, "", -1);
 47 }
