1 /**	@class Method to turn an entity into a simple finite state machine.
  2 	@description Allows you to give any game entity multiple update, draw, and event handling routines, and switch between them quickly and easily. Furnishes your entity with a new method "set_state(name)" which sets the state and methods to that name. For example, if you have a method running_draw() and a method standing_draw(), when you call set_state("running") the running_draw() method will be set as the main draw() method. This is true of any method prefixed with "running_" e.g. running_update() and running_keyDown_37() for example.
  3 	@param entity is the game entity you want to turn into a finite state machine.
  4 */
  5 function statemachine(entity) {
  6 	entity.state = "";
  7 	entity.set_state = function(newstate) {
  8 		// loop through every method and find the ones we want to change to
  9 		for (var m in this) {
 10 			if (!m.indexOf(newstate + "_") && typeof(this[m]) === "function") {
 11 				var parts = m.split("_");
 12 				if (parts.length == 2) {
 13 					this[parts[1]] = this[m];
 14 				}
 15 			}
 16 		}
 17 		this.state = newstate;
 18 	}
 19 	return entity;
 20 }
