  9 /**	@class The core jsGameSoup library for entity and event management. You can either pass your own canvas like this: var gs = new JSGameSoup(mycanvas, 30); myGameFunction(gs); gs.launch(); or else when the jsgamesoup.js script is loaded, it will attach a `new JSGameSoup()` instantiation to every canvas tag which has an attribute 'jsgs'. The attribute 'jsgs' specifies the name of the function which should be called to launch the game script associated with that canvas. The 'fps' attribute specifies the desired frame rate of the game engine for that canvas. Once the jsGameSoup engine has been attached to the canvas it starts running immediately. The jsGameSoup engine keeps a list of objects to update and draw every frame. In order to make things happen in your game, you should create objects and add them to the engine with the addEntity() method.
 10 	@param canvas can be a canvas element object, or the ID of the canvas element which this instance of JSGameSoup should attach itself to. If you pass another type of object, for example a <div> tag, a canvas of the same size will be created automatically.
 11 	@param framerate The number of frames per second the game will try to run at on this canvas.
 12 */

115 	/** Schedule a callback to be run after some number of frames.
116 		@param frames is the number of frames to wait before running this callback. Must be > 1.
117 		@param callback is the callback to be run when the given number of frames has elapsed.
118 	*/
119 	this.schedule = function(frames, callback) {
120 		if (frames > 0) {
121 			var when = this.frameCount + frames;
122 			if (!(when in scheduled)) {
123 				scheduled[when] = [];
124 			}
125 			scheduled[when].push(callback);
126 		}
127 	}

144 	/** 	Draw a polygon
145 		@param poly A list of 2-element lists (x,y) like [[x1, y1,], [x2, y2]...]
146 		@param open Whether this polygon is closed or not
147 		@tag graphics
148 	*/
149 	this.polygon = function polygon(poly, open) {
150 		this.ctx.save();
151 		this.ctx.beginPath();
152 		this.ctx.moveTo(poly[0][0], poly[0][1]);
153 		for (var n = 0; n < poly.length; n++) {
154 			this.ctx.lineTo(poly[n][0], poly[n][1]);
155 		}
156 		if (open)
157 			this.ctx.lineTo(poly[0][0], poly[0][1]);
158 		this.ctx.closePath();
159 		this.ctx.fill();
160 		this.ctx.stroke();
161 		this.ctx.restore();
162 	}

181 		Returns a random real number between start and end. You should use the random.js with seedable random numbers instead.
182 		@param start The lower bound of the real number to be chosen.
183 		@param end The upper bound of the real number to be chosen.

187 		return Math.random() * (end - start) + start;

192 		@param a The first point.
193 		@param b The second point.

196 	this.distance = function distance(a, b) {

251 	// get the position of the triggered event
252 	this.getSetPointerPosition = function getSetPointerPosition(ev) {
253 		// was this a touch?
254 		if (ev.touches) {
255 			var touch = ev.touches[0];
256 			mouseX = touch.clientX - canvas.offsetLeft;
257 			mouseY = touch.clientY - canvas.offsetTop;
258 			this.pointerPosition = [mouseX, mouseY];
259 			return this.pointerPosition;
260 		} else {
261 			// Get the mouse position relative to the canvas element.
262 			if (ev.layerX || ev.layerX == 0) { // Firefox
263 				mouseX = ev.layerX - canvas.offsetLeft;
264 				mouseY = ev.layerY - canvas.offsetTop;
265 			} else if (ev.offsetX || ev.offsetX == 0) { // Opera
266 				mouseX = ev.offsetX;
267 				mouseY = ev.offsetY;
268 			} else {
269 				mouseX = ev.clientX - canvas.offsetLeft + scrollX;
270 				mouseY = ev.clientY - canvas.offsetTop + scrollY;
271 			}
272 			this.pointerPosition = [mouseX, mouseY];
273 			return this.pointerPosition;
274 		}
275 	}

449 	this.addEntityToSpecialistLists = function addEntityToSpecialistLists(e) {
450 		// add this object to any specialist list to which it belongs
451 		for (var method in e) {
452 			// if this entity has an event listener for when certain keys are held down
453 			if (method.indexOf("keyHeld") == 0 && entitiesKeyHeld.indexOf(e) < 0) {

455 			}
456 			// if this entity has any type of collision detection happening
457 			if (method.indexOf("collision") == 0 && entitiesColliders.indexOf(e) < 0) {

459 			}
460 		}

473 	/** This is our main game loop, which gets launched automatically with the launch() method. */
474 	this.gameSoupLoop = function gameSoupLoop() {
475 		// do we have any scheduled callbacks waiting to run?
476 		if (this.frameCount in scheduled) {
477 			for (var s=0; s<scheduled[this.frameCount].length; s++) {
478 				// run the callback
479 				scheduled[this.frameCount][s](this);
480 			}
481 			delete scheduled[this.frameCount];
482 		}
483 		
484 		// run .update() on every entity in our list
485 		for (var o=0; o<entities.length; o++) {
486 			if (entities[o].update) {
487 				entities[o].update(this);
488 			}
489 		}
490 		
491 		// get all the events out of the event queue and execute the event method on it's entity
492 		var ev = null;
493 		entitiesTriggered = [];
494 		while (ev = entityEventQueue.pop()) {
495 			if (ev[0][ev[1]](ev[2])) {
496 				entitiesTriggered.push(ev[0]);
497 			}
498 		}
499 		
500 		// add any new entities which the user has added
501 		for (var o=0; o<addEntities.length; o++) {
502 			// TODO: make sublists of drawables to make the loops tighter
503 			// TODO: make sublists of updateables to make the loops tighter
504 			// TODO: make sublists of event handling entities
505 			// set default priority if it's not set
506 			if (!addEntities[o].priority) addEntities[o].priority = 0;
507 			// this hack is so that entities get a stable sort because WebKit's sort is not stable, gah!
508 			addEntities[o].priority_preference = totalEntityCount;
509 			totalEntityCount += 1;
510 			// add the new one to our list of entities
511 			entities.push(addEntities[o]);
512 			this.addEntityToSpecialistLists(addEntities[o]);
513 			if (addEntities[o].init) {
514 				addEntities[o].init(this);
515 			}
516 			if (addEntities[o].update) {
517 				addEntities[o].update(this);
518 			}
519 		}
520 		// sort entities by priority after adding new entities
521 		if (addEntities.length)	this.sortEntities();
522 		addEntities = [];
523 		
524 		// delete any entities the user has asked to remove
525 		for (var o=0; o<delEntities.length; o++) {
526 			entities.remove(delEntities[o]);
527 			this.removeEntityFromSpecialistLists(delEntities[o]);
528 		}
529 		delEntities = [];
530 		
531 		// test for held keys and send them to listening entities
532 		for (var o=0; o<entitiesKeyHeld.length; o++) {
533 			var hasHeld = false;
534 			for (var k in this.heldKeys) {
535 				if (this.heldKeys[k]) {
536 					this.callAll(entitiesKeyHeld, "keyHeld_" + k);
537 					hasHeld = true;
538 				}
539 			}
540 			if (hasHeld)
541 				this.callAll(entitiesKeyHeld, "keyHeld");
542 		}
543 		
544 		// clear the background
545 		this.clear();
546 		// run .draw() on every entity in our list
547 		for (var o=0; o<entities.length; o++) {
548 			if (entities[o].draw) {
549 				this.ctx.save();
550 				entities[o].draw(this.ctx, this);
551 				this.ctx.restore();
552 			}
553 		}
554 		
555 		this.frameCount += 1;
556 	}

558 	/** Launch an instance of jsGameSoup (generally happens automatically). */
559 	this.launch = function launch() {
560 		var GS = this;
561 		// launch our custom loop
562 		if (navigator.userAgent.indexOf("MSIE") == -1) {
563 			var looping = setInterval(function() {
564 				try {
565 					GS.gameSoupLoop();
566 				} catch(e) {
567 					clearInterval(looping);
568 					throw(e);
569 				}
570 			}, 1000 / this.framerate);
571 		} else {
572 			// internet explorer is too hard to debug with try/catch as it forgets the stack :(
573 			var looping = setInterval(function() { GS.gameSoupLoop(); }, 1000 / this.framerate);
574 		}
575 		// DEBUG:
576 		//setInterval(function() { for (var e=0; e<entities.length; e++) console.log(entities[e].x + ", " + entities[e].y); }, 1000);
577 		//setInterval(function() { console.log(entities.length) }, 1000);
578 		//setInterval(function() { console.log(entitiesColliders.length) }, 1000);
579 	}

582 		Call this after any entity's priority changes - it's automatically called when new entities are added.
583 	*/
584 	this.sortEntities = function() {
585 		entities.sort(function(a, b) { return a.priority == b.priority ? a.priority_preference - b.priority_preference : a.priority - b.priority; });

593 	/* Detect whether a point is inside a polygon (list of points) or not */
594 	this.pointInPoly = function pointInPoly(pos, poly) {
595 		/* This code is patterned after [Franklin, 2000]
596 		http://www.geometryalgorithms.com/Archive/algorithm_0103/algorithm_0103.htm
597 		Tells us if the point is in this polygon */
598 		cn = 0
599 		pts = poly.slice();
600 		pts.push([poly[0][0], poly[0][1]]);
601 		for (var i=0; i<poly.length; i++)
602 			if (((pts[i][1] <= pos[1]) && (pts[i+1][1] > pos[1])) || ((pts[i][1] > pos[1]) && (pts[i+1][1] <= pos[1])))
603 				if (pos[0] < pts[i][0] + (pos[1] - pts[i][1]) / (pts[i+1][1] - pts[i][1]) * (pts[i+1][0] - pts[i][0]))
604 		cn += 1
605 		return cn % 2
606 	}

608 	/* Detect whether a point is inside a box or not */
609 	this.pointInBox = function pointInBox(pos, box) {
610 		return pos[0] >= box[0] && pos[0] <= box[2] && pos[1] >= box[1] && pos[1] <= box[3];
611 	}
612 	
613 	/* Detect whether a point is inside a circle */
614 	this.pointInCircle = function pointInCircle(pos, circle) {
615 		return this.distance(pos, circle.slice(0,2)) <= circle[2];
616 	}

622 	// call a method on an entity if the point is inside the entity's polygon/circle/box
623 	// used in mouse events to send mouseDown and mouseUp events into the entity
624 	this.pointInEntitiesCall = function pointInEntitiesCall(pos, fn, arg) {
625 		var hit = [];
626 		for (var e=0; e<entities.length; e++) {
627 			var ent = entities[e];
628 			if (ent[fn]) {
629 				if ((ent.pointerPoly && this.pointInPoly(pos, ent.pointerPoly())) || (ent.pointerBox && this.pointInBox(pos, ent.pointerBox())) || (ent.pointerCircle && this.pointInCircle(pos, ent.pointerCircle())) || (ent.pointerTest && ent.pointerTest(pos))) {
630 					entityEventQueue.push([ent, fn, arg]);
631 					hit.push(ent);
632 				}
633 			}
634 		}
635 		return hit;
636 	}

639 	// used for key events etc.
640 	this.entitiesCall = function entitiesCall(fn, arg) {
641 		this.callAll(entities, fn, arg);

646 		for (var e=0; e<arr.length; e++) {
647 			if (arr[e][fn]) {
648 				entityEventQueue.push([arr[e], fn, arg]);
649 			}
650 		}
651 	}

659 /**
660  *	Helper function which is automatically called to launch JSGameSoup on a canvas.
661  *
662  *	Finds all canvas tags in the document,
663  *	calls the function named in the attribute 'jsgs',
664  *	passes a new JSGameSoup instance to that function,
665  *	fps is set in the attribute called 'fps'.
666  *
667  *	Modified version of processingjs' init.js example script.
668  *
669  */
670 
671 function FindAndLaunchCanvasJSGS() {
672 	var canvases = document.getElementsByTagName("canvas");
673 	for ( var i = 0; i < canvases.length; i++ ) {
674 		var launchfn = null;
675 		var launchfps = 15;
676 		if (canvases[i].getAttribute) {
677 			if (canvases[i].getAttribute('jsgs'))
678 				launchfn = canvases[i].getAttribute('jsgs');
679 			if (canvases[i].getAttribute('fps'))
680 				launchfps = canvases[i].getAttribute('fps');
681 		} else {
682 			if (canvases[i].jsgs)
683 				launchfn = canvases[i].jsgs;
684 			if (canvases[i].fps)
685 				launchfps = canvases[i].fps;
686 		}
687 		if (launchfn) {
688 			var gs = new JSGameSoup(canvases[i], launchfps);
689 			this[launchfn](gs);
690 			gs.launch();
691 		}
692 	}
693 }
