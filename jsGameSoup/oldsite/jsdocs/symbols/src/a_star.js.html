 27 /**
 28 	@class AStar path finding algorithm implementation by Benjamin Hardin.
 29 	@description Finds a path between two different squares on a two dimensional board of squares with some squares marked as impassable. Original MIT licensed source code from http://46dogs.blogspot.com/2009/10/star-pathroute-finding-javascript-code.html
 30 	@param initial_board is a two dimensional array of 1/0 values for passable/non-passable squares.
 31 */
 32 function AStar(initial_board) {
 33 	var board = initial_board;
 34 	
 35 	/**	@method 
 36 		@description Finds a short path between two points on a 2 dimensional board.
 37 		@param start is a two-value array of x, y position e.g. [5, 4]
 38 		@param destination is a two-value array of x, y position e.g. [5, 4]
 39 	*/
 40 	this.find_path = function(start, destination)
 41 	{
 42 		var columns = board.length;
 43 		var rows = board[0].length;
 44 		// Create start and destination as true nodes
 45 		start = new this.Node(start[0], start[1], -1, -1, -1, -1);
 46 		destination = new this.Node(destination[0], destination[1], -1, -1, -1, -1);
 47 		
 48 		var open = []; // List of open nodes (nodes to be inspected)
 49 		var closed = []; // List of closed nodes (nodes we've already inspected)
 50 		
 51 		var g = 0; // Cost from start to current node
 52 		var h = this.heuristic(start, destination); // Cost from current node to destination
 53 		var f = g+h; // Cost from start to destination going through the current node
 54 		
 55 		// Push the start node onto the list of open nodes
 56 		open.push(start); 
 57 		
 58 		//Keep going while there's nodes in our open list
 59 		while (open.length > 0)
 60 		{
 61 			// Find the best open node (lowest f value)
 62 			
 63 			// Alternately, you could simply keep the open list sorted by f value lowest to highest,
 64 			// in which case you always use the first node
 65 			var best_cost = open[0].f;
 66 			var best_node = 0;
 67 			
 68 			for (var i = 1; i < open.length; i++)
 69 			{
 70 				if (open[i].f < best_cost)
 71 				{
 72 					best_cost = open[i].f;
 73 					best_node = i;
 74 				}
 75 			}
 76 			
 77 			// Set it as our current node
 78 			var current_node = open[best_node];
 79 			
 80 			// Check if we've reached our destination
 81 			if (current_node.x == destination.x && current_node.y == destination.y)
 82 			{
 83 				var path = [destination]; //Initialize the path with the destination node
 84 
 85 				// Go up the chain to recreate the path 
 86 				while (current_node.parent_index != -1)
 87 				{
 88 					current_node = closed[current_node.parent_index];
 89 					path.unshift(current_node);
 90 				}
 91 
 92 				return path;
 93 			}
 94 			
 95 			// Remove the current node from our open list
 96 			open.splice(best_node, 1);
 97 			
 98 			// Push it onto the closed list
 99 			closed.push(current_node);
100 			
101 			// Expand our current node (look in all 8 directions)
102 			for (var new_node_x = Math.max(0, current_node.x-1); new_node_x <= Math.min(columns-1, current_node.x+1); new_node_x++)
103 				for (var new_node_y = Math.max(0, current_node.y-1); new_node_y <= Math.min(rows-1, current_node.y+1); new_node_y++)
104 				{
105 					if (board[new_node_x][new_node_y] == 0 // If the new node is open
106 						|| (destination.x == new_node_x && destination.y == new_node_y)) //or the new node is our destination
107 					{
108 						// See if the node is already in our closed list. If so, skip it.
109 						var found_in_closed = false;
110 						for (var i in closed)
111 							if (closed[i].x == new_node_x && closed[i].y == new_node_y)
112 							{
113 								found_in_closed = true;
114 								break;
115 							}
116 						
117 						if (found_in_closed)
118 							continue;
119 						
120 						// See if the node is in our open list. If not, use it.
121 						var found_in_open = false;
122 						for (var i in open)
123 							if (open[i].x == new_node_x && open[i].y == new_node_y)
124 							{
125 								found_in_open = true;
126 								break;
127 							}
128 						
129 						if (!found_in_open)
130 						{
131 							var new_node = new this.Node(new_node_x, new_node_y, closed.length-1, -1, -1, -1);
132 
133 							new_node.g = current_node.g + Math.floor(Math.sqrt(Math.pow(new_node.x-current_node.x, 2)+Math.pow(new_node.y-current_node.y, 2)));
134 							new_node.h = this.heuristic(new_node, destination);
135 							new_node.f = new_node.g+new_node.h;
136 
137 							open.push(new_node);
138 						}
139 					}
140 				}
141 		}
142 
143 		return [];
144 	}
145 	
146 	/**
147 		@method
148 		@description Update the board to a new configuration.
149 		@param new_board is the new board data - a two dimensional array.
150 	*/
151 	this.update_board = function(new_board) {
152 		board = new_board
153 	}
154 	
155 	/**	
156 		@method
157 		@description Function that defines the heuristic for closeness of two nodes. Override this to define your own heuristic for distance between two nodes. An A* heurisitic must be admissible, meaning it must never overestimate the distance to the goal. In other words, it must either underestimate or return exactly the distance to the goal.
158 		@param current_node is the source node we are measuring a path from in the current iteration.
159 		@param destination_node is the endpoint of the current path being tested.
160 	*/
161 	this.heuristic = function(current_node, destination_node) {
162 		var x = current_node.x - destination_node.x;
163 		var y = current_node.y - destination_node.y;
164 		return x * x + y * y;
165 	}
166 	
167 	/* Representation of a node being tested in the 2d board. */
168 	this.Node = function(x, y, parent_index, g, h, f) {
169 		this.x = x;
170 		this.y = y;
171 		this.parent_index = parent_index;
172 		this.g = g;
173 		this.h = h;
174 		this.f = f;
175 	}
176 }
