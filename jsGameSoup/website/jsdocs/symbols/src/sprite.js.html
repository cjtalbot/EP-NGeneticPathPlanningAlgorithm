1 /**
  2  	@class A sprite class with different image/animation sequences based upon defined 'actions'.
  3 	@description Allows you to associate a set of animations with a particular entity. To do vector graphics, use the canvas tag methods instead.
  4 	@param anchor selects which side of the sprite's rectangle to 'anchor' the animation to. e.g. ["center", "bottom"] will anchor the sprite to the ground (side view) whilst ["right", "center"] would anchor it to the right hand side.
  5 	@param frames is a dictionary containing all actions and their associated set of images and the number of frames to show each image for. For instance: {"stand": [["img/stand.png", 0],], "walk": [["img/walk1.png", 3], ["img/walk2.png", 3],]} where each walk frame is shown for three frames.
  6 	@param loadedcallback is a function that is called once all of the frames in all action animations are successfully loaded.
  7 */
  8 function Sprite(anchor, frames, loadedcallback) {
  9 	var loadcount = 0;
 10 	var action = "";
 11 	var framecount = -1;
 12 	var frame = 0;
 13 	var sprite = this;
 14 	var numframes = 0;
 15 	var loopcallback = null;
 16 	this.loaded = false;
 17 	this.width = 0;
 18 	this.height = 0;
 19 	
 20 	// load up all of the images
 21 	for (var a in frames) {
 22 		// replace string entries with Images, unless they already are
 23 		for (var f=0; f<frames[a].length; f++) {
 24 			if (typeof(frames[a][f][0]) == "string") {
 25 				loadcount += 1;
 26 				var img = new Image();
 27 				img.src = frames[a][f][0];
 28 				frames[a][f][0] = img;
 29 				img.onload = function () {
 30 					loadcount -= 1;
 31 					if (loadcount == 0) {
 32 						sprite.width = parseInt(img.width);
 33 						sprite.height = parseInt(img.height);
 34 						sprite.loaded = true;
 35 						if (loadedcallback) {
 36 							loadedcallback();
 37 						}
 38 					}
 39 				}
 40 			}
 41 		}
 42 		if (loadcount == 0) {
 43 			// they passed in images. set our width and height to the first image
 44 			if (frames[a].length && frames[a][0].length) {
 45 				sprite.width = frames[a][0][0].width;
 46 				sprite.height = frames[a][0][0].height;
 47 			}
 48 			sprite.loaded = true;
 49 			if (loadedcallback) {
 50 				loadedcallback();
 51 			}
 52 		}
 53 	}
 54 	
 55 	// calculate offsets (center, right, left, top, bottom)
 56 	var calc_x = {
 57 		left: function(frame) {
 58 			return 0;
 59 		},
 60 		right: function(frame) {
 61 			return frame.width;
 62 		},
 63 		center: function(frame) {
 64 			return frame.width / 2;
 65 		}
 66 	}
 67 	
 68 	var calc_y = {
 69 		top: function(frame) {
 70 			return 0;
 71 		},
 72 		bottom: function(frame) {
 73 			return frame.height;
 74 		},
 75 		center: function(frame) {
 76 			return frame.height / 2;
 77 		}
 78 	}
 79 	
 80 	/**
 81 		Sets which named animation/action to play.
 82 		@param a is the name of the animation/action you defined on initialisation.
 83 		@param reset indicates whether the frame number should be reset to the start of the animation.
 84 		@param callback is called when the animation has completed one loop - receives parameter "action".
 85 	**/
 86 	this.action = function(a, reset, callback) {
 87 		if (typeof(callback) == "undefined") {
 88 			loopcallback = null;
 89 		} else {
 90 			loopcallback = callback;
 91 		}
 92 		action = a;
 93 		numframes = frames[a].length;
 94 		if (reset) {
 95 			framecount = frames[a][0][1];
 96 			frame = 0
 97 		} else {
 98 			frame = frame % frames[a].length;
 99 		}
100 		sprite.update = sprite._update;
101 		sprite.draw = sprite._draw;
102 		sprite.aabb = sprite._aabb;
103 		return this;
104 	}
105 	
106 	/** Returns the current action being played. **/
107 	this.get_action = function() {
108 		return action;
109 	}
110 	
111 	/** Returns the current frame number being played. **/
112 	this.get_frame = function() {
113 		return frame;
114 	}
115 	
116 	/** Returns the total number of frames. **/
117 	this.get_num_frames = function() {
118 		return numframes;
119 	}
120 	
121 	/** Sets the animation frame to play **/
122 	this.set_frame = function(newframe) {
123 		frame = newframe;
124 	}
125 	
126 	// increment frame counter etc.
127 	this._update = function() {
128 		framecount -= 1;
129 		if (framecount <= 0) {
130 			if (loopcallback && (frame + 1 >= frames[action].length)) {
131 				loopcallback(action);
132 			}
133 			frame = (frame + 1) % frames[action].length;
134 			var fimg = frames[action][frame]
135 			framecount = fimg[1];
136 			this.width = fimg[0].width;
137 			this.height = fimg[0].height;
138 		}
139 	}
140 	
141 	// draw this sprite on canvas c at position with respect to the anchor specified
142 	this._draw = function(c, pos) {
143 		var i = frames[action][frame][0];
144 		c.drawImage(i, pos[0] - calc_x[anchor[0]](i), pos[1] - calc_y[anchor[1]](i));
145 	}
146 	
147 	// returns the axis-aligned bounding-box of this sprite	for the current frame
148 	this._aabb = function(pos) {
149 		var i = frames[action][frame][0];
150 		return [pos[0] - calc_x[anchor[0]](i), pos[1] - calc_y[anchor[1]](i), i.width, i.height];
151 	}
152 	
153 	/** Call this method from inside the owner entity's update() method. */
154 	this.update = function() {};
155 	
156 	/** 
157 		Draw the sprite. Call this method from inside the owner entity's draw() method.
158 		@param c is the canvas to draw on (passed to the entity's draw(c) method)
159 		@param pos is the position to draw at relative to the anchor point.
160 	 **/
161 	this.draw = function() {};
162 	
163 	/**
164 		Returns the axis aligned bounding box of this sprite at its current frame.
165 		@param pos is the position to get the aabb relative to (factors the anchor point in too).
166 	**/
167 	this.aabb = function() { return [0, 0, 0, 0]; };
168 }
169 
170 /**
171 	@method preload
172 	@description Pre-loads a whole array of images. Provides feedback on which images are loaded via the progresscallback, which returns the number of images left to load each time one is loaded, and 0 when the final image is loaded.
173 	@param images is an array of strings containing the URLs of images to load.
174 	@param completedcallback is a function which is called when all images are loaded
175 	@param progresscallback is a function accepting an integer, which is the count of images left to load
176 */
177 Sprite.preload = function(images, completedcallback, progresscallback) {
178 	var loadcount = images.length;
179 	var img = [];
180 	for (var i=0; i<images.length; i++) {
181 		img[i] = new Image();
182 		img[i].onload = function () {
183 			loadcount -= 1;
184 			if (progresscallback)
185 				progresscallback(loadcount);
186 			if (loadcount == 0 && completedcallback) {
187 				completedcallback();
188 			}
189 		}
190 		img[i].src = images[i];
191 	}
192 }
