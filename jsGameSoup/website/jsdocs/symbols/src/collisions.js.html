1 /** 
  2 	@namespace
  3 	Test for collisions between arrays of entities. Each collision method accept two arrays of entities. Every entity in the first array will be tested for collisions against every entity in the second array. The two arrays can also be the same to test every entity against every other. When a collision is found between two entities, the collide_xxxx() method will be called on each entity involved in the collision, where "xxxx" is the type of collision (e.g. collide_aabb(), collide_circle(), collide_polygon()). The first argument is the other entity in the collision, and the second argument is the result returned from the collision, which depends on the type of collision (e.g. might be penetration depth or points or just boolean).
  4 */
  5 
  6 collide = {}
  7 
  8 // generic function for running fn on all of groupa and groupb
  9 collide.collideall = function(fn, calltype) {
 10 	var fnname = "collide_" + calltype;
 11 	return function(groupa, groupb) {
 12 		for (var a in groupa) {
 13 			for (var b in groupb) {
 14 				var ae = groupa[a];
 15 				var be = groupb[b];
 16 				if (ae != be && groupa.hasOwnProperty(a) && groupb.hasOwnProperty(b)) {
 17 					var collisionresult = fn(ae, be);
 18 					if (collisionresult) {
 19 						if (ae[fnname])
 20 							ae[fnname](be, collisionresult);
 21 						if (be[fnname])
 22 							be[fnname](ae, collisionresult);
 23 					}
 24 				}
 25 			}
 26 		}
 27 	}
 28 }

 30 /**
 31 	Axis-aligned bounding-box collision between two arrays of entities. This expects entities to have a method called get_collision_aabb() which should return a rectangle of the boundaries of the entity with the form [x, y, w, h].
 32 	@param a is an array of entities
 33 	@param b is an array of entities that will be collided with those in group a
 34 */
 35 collide.aabb = function() {};
 36 
 37 collide.collide_aabb_entities = function(a, b) {
 38 	if (b.get_collision_aabb && a.get_collision_aabb) {
 39 		var aaabb = a.get_collision_aabb();
 40 		var baabb = b.get_collision_aabb();
 41 		
 42 		return (!(aaabb[0] > baabb[0] + baabb[2] || 
 43 		baabb[0] > aaabb[0] + aaabb[2] || 
 44 		aaabb[1] > baabb[1] + baabb[3] || 
 45 		baabb[1] > aaabb[1] + aaabb[3]));
 46 	}
 47 }
 48 
 49 collide.aabb = collide.collideall(collide.collide_aabb_entities, "aabb");

 51 /**
 52 	Circle collision test between two groups of entities. This expects entities to have a method called get_collision_circle() which should return the center of the circle and the radius like this: return [[x, y], r].
 53 	@param a is an array of entities
 54 	@param b is an array of entities that will be collided with those in group a
 55 */
 56 collide.circles = function() {};
 57 
 58 collide.collide_circle_entities = function(a, b) {
 59 	if (b.get_collision_circle && a.get_collision_circle) {
 60 		var ca = a.get_collision_circle();
 61 		var cb = b.get_collision_circle();
 62 		return Math.pow(ca[0][0] - cb[0][0],2) + Math.pow(ca[0][1] - cb[0][1],2) < Math.pow(ca[1] + cb[1], 2);
 63 	}
 64 }

 66 collide.circles = collide.collideall(collide.collide_circle_entities, "circle");

 68 /**
 69 	Helper function which tests whether a point is within a particular polygon.
 70 	@param pos is a point of the form [x, y]
 71 	@param poly is a polygon defined as a list of points of the form [[x1, y1], [x2, y2], ... [xn, yn]]
 72  */
 73 var pointInPoly = function(pos, poly) {
 74 	/* This code is patterned after [Franklin, 2000]
 75 	http://www.geometryalgorithms.com/Archive/algorithm_0103/algorithm_0103.htm
 76 	Tells us if the point is in this polygon */
 77 	cn = 0
 78 	pts = poly.slice();
 79 	pts.push([poly[0][0], poly[0][1]]);
 80 	for (var i=0; i<poly.length; i++)
 81 		if (((pts[i][1] <= pos[1]) && (pts[i+1][1] > pos[1])) || ((pts[i][1] > pos[1]) && (pts[i+1][1] <= pos[1])))
 82 			if (pos[0] < pts[i][0] + (pos[1] - pts[i][1]) / (pts[i+1][1] - pts[i][1]) * (pts[i+1][0] - pts[i][0]))
 83 	cn += 1
 84 	return cn % 2
 85 }

 87 /**
 88 	Helper function which tests whether two lines intersect.
 89 	@param l1 is a line of the form [[x1, y1], [x2, y2]]
 90 	@param l2 is a line of the form [[x1, y1], [x2, y2]]	
 91 */
 92 var lineOnLine = function(l1, l2) {
 93 	// Detects the intersection of two lines
 94 	//   http://www.kevlindev.com/gui/math/intersection/Intersection.js
 95 	var a1 = l1[0];
 96 	var a2 = l1[1];
 97 	var b1 = l2[0];
 98 	var b2 = l2[1];
 99 	var a1x = a1[0];
100 	var a1y = a1[1];
101 	var a2x = a2[0];
102 	var a2y = a2[1];
103 	var b1x = b1[0];
104 	var b1y = b1[1];
105 	var b2x = b2[0];
106 	var b2y = b2[1];
107 	
108 	var ua_t = (b2x - b1x) * (a1y - b1y) - (b2y - b1y) * (a1x - b1x);
109 	var ub_t = (a2x - a1x) * (a1y - b1y) - (a2y - a1y) * (a1x - b1x);
110 	var u_b  = (b2y - b1y) * (a2x - a1x) - (b2x - b1x) * (a2y - a1y);
111 	
112 	if (u_b) {
113 		var ua = ua_t / u_b;
114 		var ub = ub_t / u_b;
115 		
116 		if (0 <= ua && ua <= 1 && 0 <= ub && ub <= 1) {
117 			// intersection
118 			return [a1x + ua * (a2x - a1x), a1y + ua * (a2y - a1y)];
119 		} else {
120 			return [];
121 		}
122 	} else {
123 		if (ua_t == 0 || ub_t == 0) {
124 			// coincident
125 			//return [line2]
126 			//this will be caught elsewhere anyway
127 			return [(a2x + a1x) / 2, (a2y + a1y) / 2];
128 		} else {
129 			// parallel
130 			return [];
131 		}
132 	}
133 }

135 /**
136 	Polygon collision tests whether two entities defined by polygons are touching. Entities should have a get_collision_poly() method which should return an array of lines of the form [[x1, y1], [x2, y2], ... [xn, yn]].
137 	@param a is an array of entities
138 	@param b is an array of entities that will be collided with those in group a
139 */
140 collide.polys = function() {};
141 
142 collide.collide_poly_entities = function(a, b) {
143 	if (b.get_collision_poly && a.get_collision_poly) {
144 		var e1 = a.get_collision_poly();
145 		var e2 = b.get_collision_poly();
146 		for (var l1=0; l1<e1.length; l1++) {
147 			for (var l2=0; l2<e2.length; l2++) {
148 				var compareline1 = [e1[l1], e1[(l1 + 1) % e1.length]];
149 				var compareline2 = [e2[l2], e2[(l2 + 1) % e2.length]];
150 				var linesresult = lineOnLine(compareline1, compareline2);
151 				if (linesresult.length) {
152 					return [compareline1, compareline2, linesresult];
153 				}
154 			}
155 		}
156 		// if we get here test if a point from each poly is inside the other
157 		return pointInPoly(e1[0], e2) || pointInPoly(e2[0], e1);
158 	}
159 }
160 
161 collide.polys = collide.collideall(collide.collide_poly_entities, "poly");
