1 /** 
  2 	@namespace
  3 	Provides an interface for making ajax requests, and for bulk loading data from a set of URLs, e.g. level or tile data.
  4 */
  5 network = {};
  6 
  7 /**
  8 	Make a single ajax request. Returns the request object. The default timeout is 30000 milliseconds. 
  9 	@param url is where to fetch data from.
 10 	@param callback gets called when the data arrives, taking arguments (response_data, http_request).
 11 	@param timeout_callback gets called when the request times out.
 12 	@param type is GET or POST (note that this may default to POST on some platforms if data is passed).
 13 	@param data is an associative array of data to send (optional).
 14 	@param timeout is how long in milliseconds to wait before registering a network request as failed (and calling timeout_callback).
 15 */
 16 network.makeRequest = function(url, callback, timeout_callback, type, data, timeout) {
 17 	var http_request = false;
 18 	var requestComplete = false;
 19 	if (!timeout) timeout=30000;
 20 	if (!type) type = "GET";
 21 	if (!data) data = null;
 22 	
 23 	// Create the XML RPC object
 24 	if (window.XMLHttpRequest) {
 25 		http_request = new XMLHttpRequest();
 26 	}
 27 	else if (window.ActiveXObject) {
 28 		// Internet explorer
 29 		http_request = new ActiveXObject("Microsoft.XMLHTTP");
 30 	}
 31 	
 32 	// When we receive a message back from an XML RPC request
 33 	http_request.onreadystatechange = function() {
 34 		if (typeof http_request != "undefined")	{
 35 			// reponse 4 = 'request complete'
 36 			if (http_request.readyState == 4) {
 37 				response = http_request.responseText
 38 				if (response != "") {
 39 					if (callback)
 40 						callback(response, http_request);
 41 				}
 42 				requestComplete = true;
 43 			}
 44 		} else {
 45 			http_request = false;
 46 			requestComplete = true;
 47 		}
 48 	};
 49 	
 50 	// asynchronous request
 51 	http_request.open(type, url, true);
 52 	
 53 	// if this is a post, send required headers
 54 	if (type == "POST") {
 55 		http_request.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
 56 		http_request.setRequestHeader("Content-length", ("" + data).length);
 57 	}
 58 	
 59 	// ie6 will default to post if any argument is passed
 60 	if (data) {
 61 		http_request.send("" + data);
 62 	} else {
 63 		http_request.send();
 64 	}
 65 	
 66 	// make sure we cancel everything if we reach the timeout
 67 	setTimeout(function() {
 68 		if (http_request.readyState != 4) {
 69 			if (caller.network_timeout)
 70 				caller.network_timeout(http_request, this);
 71 			http_request = false;
 72 			requestComplete = true;
 73 		}
 74 	}, timeout);
 75 };

 77 /**
 78 	Bulk load data, e.g. tile or level data, from urls. Loads an array of urls and calls progresscallback(number_of_urls_left, data_received, url, all_data_loaded_from_all_calls) each time a url has finished loading. The value in number_of_urls_left is the number of urls still to be loaded, whilst data_received is the actual data loaded from a url, and url is that url. Finally, received is an associative array where the key is the url, and the value is the content at that url.
 79 	@param urls is an array of urls to fetch
 80 	@param progresscallback is called each time a url is successfully fetched - see above for the meaning of the arguments.
 81 	@param timeoutcallback is called if a request times out.
 82 	@param timeout is the number of milliseconds to wait before timing out the request.
 83 */
 84 network.bulkLoad = function(urls, progresscallback, timeoutcallback, timeout) {
 85 	var loadcount = 0;
 86 	var received = {};
 87 	for (var i=0; i<urls.length; i++) {
 88 		// construct the closure protected version of the callback function
 89 		var callbackfunc = function(inc) {
 90 			return function (response) {
 91 				received[urls[inc]] = response;
 92 				loadcount += 1;
 93 				if (progresscallback)
 94 					progresscallback(urls.length - loadcount, response, urls[inc], received);
 95 			}
 96 		}(i);
 97 		network.makeRequest(urls[i], callbackfunc, timeoutcallback, timeout);
 98 	}
 99 }
